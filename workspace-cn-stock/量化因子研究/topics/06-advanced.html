# 示例：LLM优化因子表达式 (信达金工方案)
# 原始: std(close, 20) → 优化: ATR加权+EMA平滑+量调整
import pandas as pd

def original_vol(close, w=20):
    return close.pct_change().rolling(w).std()

def enhanced_vol(high, low, close, volume, w=20):
    """LLM优化后: ATR加权 + EMA平滑 + 量调整"""
    tr = pd.concat([high - low, (high - close.shift(1)).abs(),
                    (low - close.shift(1)).abs()], axis=1).max(axis=1)
    atr = tr.ewm(span=w).mean()
    vol_adj = (volume / volume.rolling(w).mean()).clip(0.5, 2.0)
    return ((atr / close) * vol_adj).ewm(span=w).mean()

5. Python实现：核心高频因子计算

import numpy as np, pandas as pd

class IntradayFactors:
    """从分钟K线计算日内因子，聚合为日频"""

    def realized_volatility(self, df):
        """已实现波动率: 5分钟收益率标准差"""
        ret_5m = df["close"].resample("5min").last().pct_change()
        return np.sqrt((ret_5m ** 2).sum())

    def realized_skewness(self, df):
        """已实现偏度: 5分钟收益率三阶矩"""
        ret = df["close"].resample("5min").last().pct_change().dropna()
        if len(ret) < 10: return np.nan
        rv = (ret ** 2).sum()
        return np.sqrt(len(ret)) * (ret ** 3).sum() / (rv ** 1.5)

    def volume_peak_count(self, df, threshold=2.0):
        """波峰因子: 成交量超均值N倍的次数"""
        return (df["volume"] > df["volume"].mean() * threshold).sum()

    def illiquidity(self, df):
        """非流动性: Amihud日内版"""
        ret = df["close"].pct_change().abs()
        amt = df["close"] * df["volume"]
        return (ret[amt > 0] / amt[amt > 0]).mean()

    def aggregate_daily(self, stock_id, date, df):
        """聚合为一行日频数据"""
        return {"stock": stock_id, "date": date,
                "rv": self.realized_volatility(df),
                "rskew": self.realized_skewness(df),
                "vpeak": self.volume_peak_count(df),
                "illiq": self.illiquidity(df)}

class CascadingStrategy:
    """层叠式: 月度选股 -> 周度排序 -> 日度执行"""
    def monthly_screen(self, all_scores, n=50):
        """基本面因子选出候选池"""
        return all_scores.nlargest(n)

    def weekly_rank(self, candidates, weekly_scores, n=15):
        """候选池内用技术因子排名"""
        return weekly_scores.loc[candidates.index].nlargest(n)

    def daily_execute(self, holdings, liquidity):
        """流动性因子决定执行时点"""
        return [{"stock": s, "action": "now" if liquidity[s] > 0.5 else "wait"}
                for s in holdings.index]

优点：逻辑清晰，每层独立，容易调试和归因。缺点：上游错误向下传导，月度层漏掉的好股票无法在后续层补回。

b) 融合式（Blending）

final_score = 0.40 × monthly + 0.35 × weekly + 0.25 × daily
class BlendingStrategy:
    """融合式: 各频率因子加权混合"""
    def __init__(self):
        self.w = {"monthly": 0.40, "weekly": 0.35, "daily": 0.25}

    def blend(self, m_scores, w_scores, d_scores):
        common = m_scores.index & w_scores.index & d_scores.index
        return (self.w["monthly"] * m_scores[common] +
                self.w["weekly"]  * w_scores[common] +
                self.w["daily"]   * d_scores[common]).sort_values(ascending=False)

    def adaptive_weights(self, ic_history, lookback=60):
        """基于近期IC动态调整权重"""
        recent = ic_history.tail(lookback).mean()
        total = recent.sum()
        return {k: max(0.1, recent[k]/total) for k in self.w} if total > 0 else {k: 1/3 for k in self.w}

优点：充分利用所有信号，容错性好。缺点：权重调优困难，需要所有频率因子同时可用。

双系统互补架构 ┌──────────────────────────┐ ┌──────────────────────────┐ │ quant_x (量化选股) │ │ 模拟交易系统 │ │ Qlib + LightGBM │ │ RuleStrategy / │ │ Alpha158 + 高频因子 │ │ ClaudeStrategy │ │ │ │ │ │ 输出: 月度/周度候选池 │ ==> │ 输出: 日度买卖+执行 │ │ 频率: 周度/月度 │ │ 频率: 日度 │ └──────────────────────────┘ └──────────────────────────┘ └──────────────┬───────────────────┘ ▼ ┌─────────────────────────┐ │ 统一风控 & 执行引擎 │ │ 组合止损/行业监控/成本 │ └─────────────────────────┘
class MultiHorizonOrchestrator:
    """多周期编排器: 连接选股系统和交易系统"""
    def __init__(self, qlib_model, trade_engine):
        self.model = qlib_model; self.engine = trade_engine
        self.pool = []; self.holdings = {}

    def weekly_update(self, date):
        """周度: Qlib模型刷新候选池"""
        pred = self.model.predict(date)
        self.pool = pred.nlargest(50).index.tolist()
        return pred.loc[self.pool].nlargest(15)

    def daily_execute(self, date, target):
        """日度: 交易引擎决定买卖"""
        to_buy = set(target.index) - set(self.holdings)
        to_sell = set(self.holdings) - set(target.index)
        orders = []
        for s in to_sell:
            if self.engine.get_sell_signal(s, date).should_execute:
                orders.append({"stock": s, "side": "sell"})
        for s in to_buy:
            if self.engine.get_buy_signal(s, date).liquidity_ok:
                orders.append({"stock": s, "side": "buy",
                               "weight": target[s]})
        return orders

    def run(self, start, end):
        """主循环: 周度选股 + 日度执行"""
        for date in pd.bdate_range(start, end):
            if date.weekday() == 0:  # 周一更新
                target = self.weekly_update(date)
            self.engine.submit_orders(self.daily_execute(date, target))
            self.holdings = self.engine.get_positions()
